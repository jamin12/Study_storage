from itertools import permutations

import math
# 소수 판별
def primenumber(x):
    if x == 0 or x == 1: return False
    for i in range (2, int(math.sqrt(x) + 1)):	# 2부터 x의 제곱근까지의 숫자
        if x % i == 0:		# 나눠떨어지는 숫자가 있으면 소수가 아님
            return False
    return True

def solution(numbers):
    answer = 0
    # 한번 소수로 검사한 숫자가 들어가 중복 검사 방지
    check = []
    numbers = list(numbers)
    for i in numbers:
        if i == "0" or i == "1":
            continue
        if int(i) in check:
            continue
        # 한자리수 검사
        if (primenumber(int(i))):
            check.append(int(i))
            answer += 1
    
    for i in range(2,len(numbers) + 1):
        # 조합을 이용해 중복은 없애고 검사
        number = list(set(permutations(numbers,i)))
        for j in range(len(number)):
            if int(''.join(number[j])) in check:
                continue
            if primenumber(int(''.join(number[j]))):
                check.append(int(''.join(number[j])))
                answer += 1
    return answer

print(solution("17"))
print(solution("011"))
print(solution("002"))
print(solution("333"))
